#define ENABLE_DEBUG

#ifdef ENABLE_DEBUG
  #define DEBUG_ESP_PORT Serial
  #define NODEBUG_WEBSOCKETS
  #define NDEBUG
#endif 

#include <Arduino.h>

#if defined(ESP8266)
  #include <ESP8266WiFi.h>
#elif defined(ESP32) || defined(ARDUINO_ARCH_RP2040)
  #include <WiFi.h>
#endif

#include "SinricPro.h"
#include "SinricProSwitch.h"

// === CONFIGURA√á√ïES DE REDE ===
#define WIFI_SSID     "##########################"
#define WIFI_PASS     "##########################"

// === CREDENCIAIS SINRIC PRO ===
#define APP_KEY       "######################################################"
#define APP_SECRET    "##########################"
#define SWITCH_ID_1   "####################"

// === CONFIGURA√á√ÉO DO REL√â ===
#define RELAYPIN_1    2
#define BAUD_RATE     19200

bool onPowerState1(const String &deviceId, bool &state) {
  Serial.printf("Device 1 turned %s\n", state ? "ON" : "OFF");
  digitalWrite(RELAYPIN_1, state ? HIGH : LOW);
  return true;
}

void setupWiFi() {
  Serial.printf("[WiFi]: Conectando");

#if defined(ESP8266)
  WiFi.setSleepMode(WIFI_NONE_SLEEP);
#elif defined(ESP32)
  WiFi.setSleep(false);
#endif

  WiFi.setAutoReconnect(true);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(250);
  }

  Serial.printf("\n[WiFi]: Conectado! IP: %s\n", WiFi.localIP().toString().c_str());
}

void setupSinricPro() {
  pinMode(RELAYPIN_1, OUTPUT);
  digitalWrite(RELAYPIN_1, LOW);

  SinricProSwitch &mySwitch1 = SinricPro[SWITCH_ID_1];
  mySwitch1.onPowerState(onPowerState1);

  SinricPro.onConnected([]() {
    Serial.println("Conectado ao SinricPro");
  });

  SinricPro.onDisconnected([]() {
    Serial.println("Desconectado do SinricPro");
  });

  SinricPro.restoreDeviceStates(true);
  SinricPro.begin(APP_KEY, APP_SECRET);
}

void setup() {
  Serial.begin(BAUD_RATE);
  Serial.println("\n\nInicializando...");
  setupWiFi();
  setupSinricPro();
}

void loop() {
  SinricPro.handle();
}
