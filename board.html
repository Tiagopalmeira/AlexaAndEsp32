⚙️ Código Fonte e Configuração da Board ESP32

main.ino - Detalhes do Código

O código a seguir deve ser carregado no seu ESP32. Certifique-se de substituir os placeholders (##########################) pelas suas credenciais reais do Wi-Fi, SinricPro App Key, App Secret e o Device ID do seu switch virtual criado no portal SinricPro.

#define ENABLE_DEBUG  // Descomente para habilitar mensagens de debug no Serial Monitor

#ifdef ENABLE_DEBUG
  #define DEBUG_ESP_PORT Serial // Define a porta serial para debug
  // #define NODEBUG_WEBSOCKETS  // Descomente para desabilitar debug específico de WebSockets
  // #define NDEBUG              // Descomente para desabilitar todas as asserções (não recomendado para desenvolvimento)
#endif 

#include <Arduino.h> // Biblioteca principal do Arduino

// Seleciona a biblioteca Wi-Fi correta dependendo da placa
#if defined(ESP8266)
  #include <ESP8266WiFi.h>
#elif defined(ESP32) || defined(ARDUINO_ARCH_RP2040)
  #include <WiFi.h>
#endif

#include "SinricPro.h"       // Biblioteca principal do SinricPro
#include "SinricProSwitch.h" // Suporte para dispositivos do tipo "Switch" no SinricPro

// === CONFIGURAÇÕES DE REDE ===
// Substitua pelos dados da sua rede Wi-Fi
#define WIFI_SSID      "SEU_NOME_DE_REDE_WIFI"
#define WIFI_PASS      "SUA_SENHA_DA_REDE_WIFI"

// === CREDENCIAIS SINRIC PRO ===
// Substitua pelas suas credenciais obtidas no portal SinricPro
#define APP_KEY        "SUA_APP_KEY_DO_SINRICPRO"
#define APP_SECRET     "SEU_APP_SECRET_DO_SINRICPRO"
#define SWITCH_ID_1    "SEU_DEVICE_ID_DO_SINRICPRO" // ID do seu switch virtual

// === CONFIGURAÇÃO DO PINO DO RELÉ/LED ===
#define RELAYPIN_1     2        // GPIO2 é comumente usado para o LED embutido em muitas placas ESP32, ou conecte seu relé/LED aqui
#define BAUD_RATE      115200   // Taxa de comunicação para o Serial Monitor (pode ser 9600, 19200, etc., mas 115200 é comum para ESP32)

// Função de Callback: Chamada quando o estado do switch é alterado via SinricPro/Alexa
bool onPowerState1(const String &deviceId, bool &state) {
  Serial.printf("Dispositivo com ID '%s' mudou para o estado: %s\n", deviceId.c_str(), state ? "LIGADO" : "DESLIGADO");
  digitalWrite(RELAYPIN_1, state ? HIGH : LOW); // Aciona o pino do relé/LED
  return true; // Indica que o comando foi processado com sucesso
}

// Configura e conecta ao Wi-Fi
void setupWiFi() {
  Serial.printf("[WiFi]: Conectando a %s ", WIFI_SSID);

#if defined(ESP8266)
  WiFi.setSleepMode(WIFI_NONE_SLEEP); // Desabilita modo sleep no ESP8266 para manter conexão
#elif defined(ESP32)
  WiFi.setSleep(false); // Desabilita modo sleep no ESP32
#endif

  WiFi.setAutoReconnect(true); // Habilita reconexão automática
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  unsigned long startTime = millis();
  while (WiFi.status() != WL_CONNECTED) {
    Serial.print(".");
    delay(250);
    if (millis() - startTime > 20000) { // Timeout de 20 segundos
      Serial.println("\n[WiFi]: Falha ao conectar!");
      return;
    }
  }

  Serial.printf("\n[WiFi]: Conectado! IP: %s\n", WiFi.localIP().toString().c_str());
}

// Configura e inicia o SinricPro
void setupSinricPro() {
  pinMode(RELAYPIN_1, OUTPUT); // Define o pino do relé/LED como saída
  digitalWrite(RELAYPIN_1, LOW);  // Garante que comece desligado

  // Associa o Device ID do SinricPro à função de callback
  SinricProSwitch &mySwitch1 = SinricPro[SWITCH_ID_1];
  mySwitch1.onPowerState(onPowerState1);

  // Callbacks para status da conexão com o servidor SinricPro
  SinricPro.onConnected([]() {
    Serial.println("Conectado ao SinricPro");
  });

  SinricPro.onDisconnected([]() {
    Serial.println("Desconectado do SinricPro");
  });

  SinricPro.restoreDeviceStates(true); // Opcional: tenta restaurar o último estado conhecido dos dispositivos do servidor
  SinricPro.begin(APP_KEY, APP_SECRET); // Inicia a conexão com o SinricPro
}

// Função de configuração principal, executada uma vez
void setup() {
  Serial.begin(BAUD_RATE); // Inicia a comunicação serial
  Serial.println("\n\nInicializando...");
  setupWiFi();       // Chama a função de configuração do Wi-Fi
  setupSinricPro();  // Chama a função de configuração do SinricPro
}

// Função de loop principal, executada repetidamente
void loop() {
  SinricPro.handle(); // Essencial para manter a comunicação com o SinricPro e processar comandos
}
